178  /* to test if read operation is correct */
179  uint32_t test_read=1;
180 
181  /* Set the offsets to access to right gpio channel (1 or 2) */
182  int gpio_tri_off;
183  int gpio_data_off;
184  unsigned long gpio_cha_interrupt_en;
185  unsigned long gpio_cha_interrupt_dis;
186  unsigned long gpio_cha_isr;
187  if(channel==GPIO_CHANNEL_1){
188  gpio_tri_off=GPIO_TRI_OFFSET;
189  gpio_data_off=GPIO_DATA_OFFSET;
190  gpio_cha_interrupt_en=CHANNEL_1_IER_ENABLE;
191  gpio_cha_interrupt_dis=CHANNEL_1_IER_DISABLE;
192  gpio_cha_isr=CHANNEL_1_ISR;
193  }
194  else{
195  gpio_tri_off=GPIO2_TRI_OFFSET;
196  gpio_data_off=GPIO2_DATA_OFFSET;
197  gpio_cha_interrupt_en=CHANNEL_2_IER_ENABLE;
198  gpio_cha_interrupt_dis=CHANNEL_2_IER_DISABLE;
199  gpio_cha_isr=CHANNEL_2_ISR;
200  }
201  /*----------------------------------------------------------*/
202 
203  /* Set GPIO_TRI register values to configure the GPIO port as input */
204  *((unsigned *)(virtual_address + gpio_tri_off)) = 255;
205 
206  while(1){
207 
208  /* Enable Global Interrupt of GPIO */
209  GPIO_Interrupt(virtual_address,GPIO_GIER_OFFSET,GPIO_GIER_ENABLE);
210  /* Enable Channel Interrut of GPIO */
211  GPIO_Interrupt(virtual_address,GPIO_IP_IER_OFFSET,gpio_cha_interrupt_en);
212 
213  /* to suspend the process on gpio peripheral */
